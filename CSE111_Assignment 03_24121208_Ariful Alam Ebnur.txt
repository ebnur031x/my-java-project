public class Reader {
    public String name;
    public int capacity;
    public String [] books;
    public int count;

    public Reader (String M){
        this.count=0;
        this.name=M;
        this.capacity=2;
        this.books = new String[capacity];
        System.out.println("A new reader is created!");
    }
    public Reader(String R, int num){
        this.count=0;
        this.name=R;
        this.capacity=num;
        this.books= new String [num];
        System.out.println("A new reader is created!");
    }
    public void readerInfo(){
        System.out.println("Name: "+ name);
        System.out.println("Capacity: "+ capacity);
        System.out.println("Books: ");
        if (count==0) {
            System.out.println("No books added yet");
        } else {
            for (int i = 0; i < count; i++) {// 0<1
                System.out.println("Book " + (i+1) + ": " + this.books[i]);
            }
        }
    }

    public void addBook (String name){
        if (count<capacity) {
            this.books[this.count] = name;
            count++;

        } else{
            System.out.println("No more capacity");
        }
    }

    public void updateCapacity(int n){
        this.capacity=n;
        System.out.println("Capacity has changed to "+n);
        String [] newBooks = new String [n];
        for (int i = 0; i < count; i++) {
            newBooks[i] = books[i];
        }
        this.books= newBooks;
    }
}

public class MoneyTracker{
    public String name;
    public double cr;
    public double add;
    public double spent;

    public MoneyTracker (){
        this.cr=0.0;
        System.out.println("A new money tracker has been launched.");
    }
    public String info (){
      return "Name: "+name+"\n"+"Current Balance: "+cr;
    }

    public void createTracker(String nam){
        this.cr =1.0;
        this.name=nam;
    }

    public void income (int n ){
        System.out.println("Balance Updated");
        cr=this.cr+n;
        this.spent=n;
    }

    public void expense (int m){
        this.add=m;
        if (m > this.cr) {
            System.out.println("No enough bal");
        } else if (m == this.cr) {
            this.cr = 0;
            System.out.println("You're broke");
        } else {
            this.cr = this.cr - m;
            System.out.println("Balance updated");
        }
    }
    public void showHistory (){
        System.out.println("Last added: "+spent);
        System.out.println("Last spent: "+add);
    }
}
